    from typing import List


def solve(N: int, M: int, C: List[int]) -> int:
    answer = 0
    for i in range(N-2):
        for j in range(i+1, N-1):
            for k in range(j+1, N):
                sum_of_cards = (C[i]+C[j]+C[k])
                if sum_of_cards <= M:
                    answer = max(answer, sum_of_cards)
    return answer
